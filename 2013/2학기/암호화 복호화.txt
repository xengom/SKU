#include <stdio.h> 
#include <string.h> 
#include <conio.h>
#include <stdlib.h>
#define SIZE 26 
int main() 
{ 
   int i, index=0, j, leng; 
   char word[30];
   char rword[30];
   char dic[SIZE][2][30] =   
   { 
      {"a", "d"},{"b", "e"},{"c", "f"},{"d", "g"},{"e", "h"},{"f", "i"},{"g", "j"},{"h", "k"}, 
      {"i", "l"},{"j", "m"},{"k", "m"},{"l", "o"},{"m", "p"},{"n", "q"},{"o", "r"},{"p", "s"}, 
      {"q", "t"},{"r", "u"},{"s", "v"},{"t", "w"},{"u", "x"},{"v", "y"},{"w", "z"},{"x", "a"}, 
      {"y", "b"},{"z", "c"} 
   }; 
   for(;;)
   {
   printf("1. 암호화\n"); 
   printf("2. 복호화\n"); 
   printf("3. 종료\n"); 
   scanf("%d", &i); 
   if(i==3)
	   exit(0);
   printf("단어를 입력 =  "); 
   scanf("%s", &word); 
   leng = strlen(word); 
   switch(i) 
   { 
      case 1: 
      { 
         for(i=0; i < leng; i++) 
         { 
            for(j=0; j < SIZE && dic[j][0][0]!=word[i];j++); 
            if (j < SIZE) 
            { 
               rword[i]=dic[j][1][0]; 
               }else{ 
               printf("없음\n"); 
               break; 
            } 
         } 
         rword[i]='\0'; 
         printf("%s\n",rword); 
      } 
       break;
      case 2: 
      { 
         for(i=0; i < leng; i++) 
         { 
            for(j=0; j < SIZE && dic[j][1][0]!=word[i];j++); 
            if (j < SIZE) 
            { 
               rword[i]=dic[j][0][0]; 
               }else{ 
               printf("없음\n"); 
               break; 
            } 
         } 
         rword[i]='\0'; 
         printf("%s\n",rword); 
      } 
      break;
   }
   }
   return 0; 
}
